    #非常简洁的查看版本历史
    $ git log --oneline
    40b8469 (HEAD -> master) rename readme.md to readmeme.md
    834aa69 更新文件内容注释
    3caf6e4 提交JS文件
    be372f4 添加样式表
    e3ca94b 添加主页和图片
    748b7c6 Add readme
    #显示最近两次的版本变动
    $git log -n2
    #最近的两次版本变动简单形式
    $git log -n2 --oneline
    #查看本地有多少分支
    $git branch -v
    #基于以前的版本创建新的分支:temp为新的分支名，40b84691a0为commit后的版本号前几位字母
    $git checkout -b temp 40b84691a0
    #工作区的内容跳过暂存区直接添加进版本库中
    $git commit -am'将工作区中的文件直接添加进版本库中'
    #查看有多少分支
    $git branch -av
    #查看当前分支的历史
    $git log
    #查看所有分支,英文状态下Q退出
    $git log --all
    #图形化显示分支历史变动
    $git log --all --graph
    $git log --all --oneline -n2
    $git log --all --oneline -n5 --graph
    #指定all的后面加temp不起作用
    $git log --oneline --all temp
    #没有all时有作用
    $git log --oneline temp
    #打开浏览器看帮助文档
    $git help --web log
    #打开图形化界面
    $gitk

    @zhangbitao同学提到可以设置快捷键，查了下，是设置别名。
    git config --global alias.[alias] "alias expansion"

    如下配置后
    git config --global alias.lg "log --color --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit"

    使用命令 git lg --all 相当于 git log --all --color --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cgreen(%cr) %C(bold blue)<%an>%Creset' --abbrev-commit

    显示结果格式如下
    $ git lg --all
    * d44b300 - (HEAD -> temp) add test (35 minutes ago) <leafwinder>
    | * f149573 - (master) rename readme (3 hours ago) <leafwinder>
    | * a27ea51 - add refering project (4 hours ago) <leafwinder>
    | * 7da772c - add js (4 hours ago) <leafwinder>
    |/
    * cf49dda - add styles (4 hours ago) <leafwinder>
    * 8a32811 - add index+image (4 hours ago) <leafwinder>
    * c7b9784 - my first commit (11 hours ago) <leafwinder>
