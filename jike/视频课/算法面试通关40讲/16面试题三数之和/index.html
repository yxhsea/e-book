    nums.sort()
    res = []
    for i, num in enumerate(nums):
    if i > 0 and nums[i] == nums[i-1]:
    continue

    new_nums = nums[i+1:]
    two_sums = self.twoSum(new_nums, -num)
    for two_sum in two_sums:
    res.append([num, new_nums[two_sum[0]], new_nums[two_sum[1]]])

    return res

    func twoSum(arr []int, target int) [2]int{
    tmpMap := map[int]int{}
    for k, v := range arr{
    if index, ok := tmpMap[target-v]; ok{
    return [2]int{index, k}
    }
    tmpMap[v] = k
    }
    return [2]int{-1,-1}
    }

    func threeSum(arr []int, target int) [3]int{
    for k, v := range arr{
    arrTmp := append([]int{}, arr[0:k]...)
    arrTmp = append(arrTmp, arr[k+1:]...)

    if res := twoSum(arrTmp, target-v); res[0]>=0{
    res0, res1 := res[0],res[1]
    //由于arrTmp是新构建的slice，比原slice少一个元素
    //所以判断返回的index是在当前元素左边还是右边
    //在右边要加一
    if res0>=k{
    res0+=1
    }
    if res1>=k{
    res1+=1
    }
    return [3]int{k, res0, res1}
    }
    }
    return [3]int{-1,-1,-1}
    }
